    }

void connect_wifly(void){
    
    pc.printf("Trying to connect...\n");
    wifly_connect.init();
    while (!wifly_connect.connect());
    pc.printf("Connected!\n");
    
    
    }

    }

void sleep_wifly_UART(void){
    
    //Start by exiting CMD
    wait(10); // ter tempo para abrir ao putty
    
    wifly.putc(101);       // e       
    wifly.putc(120);       // x   
    wifly.putc(105);       // i    
    wifly.putc(116);       // t   
    wifly.putc(13);        // carriage return    
   
    pc.printf("CMD exited!\n");
    wait(2);
    
    
    
    // Enter command mode
    
    wifly.putc(36);       // $       
    wifly.putc(36);       // $   
    wifly.putc(36);       // $   
     
     
    pc.printf("CMD active!\n");
    
    wait(5);
    
  
    
    wifly.putc(115); //s
    fflush(stdout);
    wifly.putc(108);  //l
    fflush(stdout);
    wifly.putc(101);  //e
    fflush(stdout);
    wifly.putc(101);  //e
    fflush(stdout);
    wifly.putc(112);  //p
    fflush(stdout);
    wifly.putc(13);  // carriage return  
    fflush(stdout);
    
    
    pc.printf("WiFly asleep!\n");
    
    }
